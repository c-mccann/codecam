public class Person(){ private String name; private int age; public Person(String name, int age){ this.name = name; this.age = age; } public String getName(){ return name; } public void setName(String name){ this.name = name; } public int getAge(){ return age; } public void setAge(int age){ this.age = age; } public static void main(String[] args){ Person carl = new Person("Carl", 23); System.out.println("Name: " + carl.getName()); System.out.println("Age: " + carl.getAge()); } }
